# https://stats.stackexchange.com/questions/595101/methods-for-measuring-non-linear-correlation

# https://towardsdatascience.com/a-new-coefficient-of-correlation-64ae4f260310/


from typing import Literal, Tuple, Union

import numpy as np
import numpy.typing as npt
from scipy import stats


def xicor(
    x: npt.ArrayLike,
    y: npt.ArrayLike,
    ties: Union[bool, Literal["auto"]] = "auto",
) -> Tuple[float, float]:
    x = np.asarray(x).flatten()
    y = np.asarray(y).flatten()
    n = len(y)

    if len(x) != n:
        raise IndexError(f"x, y length mismatch: {len(x)}, {len(y)}")

    if ties == "auto":
        ties = len(np.unique(y)) < n
    elif not isinstance(ties, bool):
        raise ValueError(
            f'expected ties either "auto" or boolean, '
            f"got {ties} ({type(ties)}) instead"
        )

    y = y[np.argsort(x)]
    r = stats.rankdata(y, method="ordinal")
    nominator = np.sum(np.abs(np.diff(r)))

    if ties:
        l = stats.rankdata(y, method="max")
        denominator = 2 * np.sum(l * (n - l))
        nominator *= n
    else:
        denominator = np.power(n, 2) - 1
        nominator *= 3

    statistic = 1 - nominator / denominator  # upper bound is (n - 2) / (n + 1)
    p_value = stats.norm.sf(statistic, scale=2 / 5 / np.sqrt(n))

    return statistic

Here, we can see that a few of the columns have some absurd values as their maximums. I mean, who needs 830 baths? That said, I'm only going to be removing outliers in the price column. This is mainly due to how robust HGB models are. Due to the binning process, or perhaps the underlying decision tree algorithm, HGB models tend to perform well even when the predictor variables contain outliers. however, I do want to reduce outliers in the target variable, both to reduce the range of values the models expects, and because some of these outliers have features similar to lower-priced houses, which could confuse the model.

For example, the following rows have fairly normal values for most of the data, with each of the features being less than or equal to the 75% value for that feature. Notably, these house listings, despite being relatively small to average, are priced at over 10,000,000 USD. Having rows that are outliers in price, but fall within normal ranges in the rest of the values could significantly lower model performance.

The value I have decided to use as the cutoff point for price is 1,000,000. This value is loosely based on being twice the 75% value. Also, most of the data falls below this value (a little over 90% of the data), so using this as the cutoff doesn't lose too much data.

Next, I want to encode categorical columns so I can use them to train the model. For encoding, I have decided to use an ordinal encoder. I originally tried to one-hot encode the data, but one-hot encoding the city column raised an error with a message along the lines of "Cannot allocate 240 GiB of memory...", so if I want to keep the city column, I'd need to use a different method. Technically, as its name implies, ordinal encoding is meant for ordered data. Therefore, for the best results, the DataFrame will be ordered by the median price for each state. This way, the states will be ordinally encoded by there order in price, which should help model performance.

from sklearn.model_selection import GridSearchCV
from time import time


def search(parameters, X_train, y_train, X_test, y_test):
    """
    Performs a grid search on a HistGradientBoostingRegressor, using split data.

    ACCEPTS:
        parameters - dict of hyperparams to seach over
        X_train - training data sans target variable
        y_train - training data target variable
        X_test - testing data sans target variable
        y_test - testing data target variable

    RETURNS:
        best_model - Optimized model based on mean_absolute_error
    """
    # Instantiate regressor
    reg = HistGradientBoostingRegressor(loss='absolute_error', random_state=42)
    
    # Start timer
    start = time()

    # Search over parameters, trying to minimize mean_absolute_error
    grid_obj = GridSearchCV(reg, parameters, scoring='neg_mean_absolute_error')
    grid_fit = grid_obj.fit(X_train, y_train)
    
    # End timer, print time taken to search
    end = time()
    print(f'{round(end-start, 4)} seconds OR\n{round((end-start)/60, 4)} minutes OR\n{round((end-start)/(60*60), 4)} hours\n')

    # Select best model
    best_model = grid_fit.best_estimator_

    # Train unoptimized model and compare
    model = reg.fit(X_train, y_train)
    preds = model.predict(X_test)
    best_preds = best_model.predict(X_test)
    
    print('Unoptimized Model\n-----------')
    print(f'Mean absolute error on testing data: {round(mean_absolute_error(y_test, preds), 2)}')
    print(f'Median absolute error on testing data: {round(median_absolute_error(y_test, preds), 2)}')
    print(f'R squared score: {round(model.score(X_test, y_test), 4)}')

    print('Optimized Model\n-----------')
    print(f'Mean absolute error on testing data: {round(mean_absolute_error(y_test, best_preds), 2)}')
    print(f'Median absolute error on testing data: {round(median_absolute_error(y_test, best_preds), 2)}')
    print(f'R squared score: {round(best_model.score(X_test, y_test), 4)}')

    return best_model

Using the above function, a search was performed on a primary set of parameters, labeled below as prim_parameters. With the hyperparameter values in the returned best_model, a secondary search was performed, where each hyperparameter used the values from the optimized model, as well as new values that were all slighlty off of the optimized value, in order to further tune the model. These parameters are labeled sec_parameters.
